/*  Animation logo  */

.animation--logo {
	width: 200px;
	height: 200px;

	span {
		position: absolute;
		top: 0;
		left: 0;
		display: inline-block;
		width: 100px;
		height: 100px;
		background: #000;
		animation: logo 8s infinite;
	}

	span:nth-child(1) {
		animation-delay: -2s;
	}

	span:nth-child(2) {
		animation-delay: -4s;
	}

	span:nth-child(3) {
		animation-delay: -6s;
	}

	span:nth-child(4) {
		animation-delay: -8s;
	}
}


@keyframes logo {
	0% {
		transform: translate(0, 0) scale(.7);
		background: #000;
	}

	25% {
		transform: translate(100px, 0) scale(.25);
		background: #000;
	}

	50% {
		transform: translate(100px, 100px) scale(1);
		background: #d12c29
	}

	75% {
		transform: translate(0, 100px) scale(.5);
		background: #d12c29
	}

	100% {
		transform: translate(0, 0) scale(.7);
		background: #000;
	}
}
